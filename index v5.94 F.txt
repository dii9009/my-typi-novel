        // 2. 붙여넣기 정규화
        // 외부(예: Claude.ai 답변 등)에서 복사하면, 화면에 보이던 "자동 줄바꿈(wrap)"
        // 위치까지 실제 개행문자(\n)로 같이 복사되는 경우가 많다.
        // 그래서 한 문장이 여러 줄로 쪼개진 채로 들어오는데, 이걸 그대로 받으면
        // 나중에 "복사" 했을 때 그 가짜 줄바꿈들이 전부 실제 줄로 처리되어 버린다.
        // -> 빈 줄(=연속 개행, 진짜 단락 구분)만 줄바꿈으로 살리고,
        //    단독 개행(=문장 중간에서 끊긴 자동 줄바꿈)은 공백으로 합쳐서
        //    원래 문장을 한 줄로 복원한다.
        editor.addEventListener('paste', (e) => {
            e.preventDefault();
            e.stopPropagation();
            let text = (e.clipboardData || window.clipboardData).getData('text/plain');

            // 1) 윈도우식 개행 통일
            text = text.replace(/\r\n/g, '\n');

            // 2) 단락 구분(빈 줄, \n\n 이상)을 임시 마커로 치환해 보호
            const PARA_MARK = '\u0000PARA\u0000';
            text = text.replace(/\n{2,}/g, PARA_MARK);

            // 3) 남은 단독 개행(문장 중간 wrap)은 공백으로 합침
            text = text.replace(/\n/g, ' ');

            // 4) 마커를 다시 단락 구분(빈 줄)으로 복원
            const paragraphs = text.split(PARA_MARK);

            // 5) Selection/Range API로 직접 DOM에 삽입 (execCommand 미사용)
            // 에디터가 완전히 비어있을 경우, range가 에디터 자체를 가리켜
            // insertNode 시 텍스트 노드가 div 밖으로 새어나가는 문제가 있다.
            // -> 에디터가 비어있으면 먼저 빈 div를 만들어 커서를 그 안에 넣어준다.
            if (editor.childNodes.length === 0) {
                const seedDiv = document.createElement('div');
                seedDiv.appendChild(document.createElement('br'));
                editor.appendChild(seedDiv);
                const seedRange = document.createRange();
                seedRange.selectNodeContents(seedDiv);
                seedRange.collapse(true);
                const seedSel = window.getSelection();
                seedSel.removeAllRanges();
                seedSel.addRange(seedRange);
            }

            const sel = window.getSelection();
            if (!sel.rangeCount) return;
            const range = sel.getRangeAt(0);
            range.deleteContents();

            const frag = document.createDocumentFragment();
            let lastNode = null;
            paragraphs.forEach(p => {
                const div = document.createElement('div');
                const trimmed = p.trim();
                if (trimmed.length > 0) {
                    div.textContent = trimmed;
                } else {
                    div.appendChild(document.createElement('br'));
                }
                frag.appendChild(div);
                lastNode = div;
            });
            range.insertNode(frag);

            // 삽입 후 에디터 최상위에 div로 감싸이지 않은 텍스트 노드나
            // 빈 텍스트 노드가 남아있다면 모두 제거해 구조를 정리한다.
            Array.from(editor.childNodes).forEach(node => {
                if (node.nodeType === Node.TEXT_NODE) {
                    if (node.textContent.trim() === '') {
                        node.remove();
                    } else {
                        const wrap = document.createElement('div');
                        wrap.textContent = node.textContent;
                        editor.replaceChild(wrap, node);
                    }
                }
            });

            // 커서를 삽입한 마지막 줄 끝으로 이동
            if (lastNode) {
                const newRange = document.createRange();
                newRange.selectNodeContents(lastNode);
                newRange.collapse(false);
                sel.removeAllRanges();
                sel.addRange(newRange);
            }

            updateWordCount();
            processSave('epi');
        });

        function handleDragOver(e) { if (e.preventDefault) e.preventDefault(); return false; }
        async function handleDrop(e) {

                const list = this.parentNode;
                const nodes = Array.from(list.children);
                const fromIdx = nodes.indexOf(dragSrcEl);
                const toIdx = nodes.indexOf(this);

            return false;
        }

        async function loadChapter(id) {

            currentChapterId = id; localStorage.setItem('lastChapterId', id);
            const snap = await getDoc(doc(db, "myNovels", id));

                titleInput.value = data.title; editor.innerHTML = data.content;
                const wasFixed = normalizeEditorStructure();
                lastSavedContent = editor.innerHTML;
                lastSaveDisplay.innerText = formatTime(data.updatedAt);
                updateWordCount(); if(window.innerWidth <= 768) closeSidebar(); renderAll();

                // 줄 구조가 깨져있던 챕터였다면, 정리된 내용을 즉시 서버에도 저장해서
                // 다음부터는 매번 정리할 필요 없이 깨끗한 상태로 불러오게 한다.
                if (wasFixed) {
                    try {
                        await setDoc(doc(db, "myNovels", id), { content: editor.innerHTML, updatedAt: Date.now() }, { merge: true });
                        lastSaveDisplay.innerText = formatTime(Date.now());
                    } catch (e) {
                        console.error('구조 자동 정리 저장 실패:', e);
                    }
                }
            }
        }

        // 과거 데이터 중 에디터 최상위에 div로 감싸이지 않은 텍스트 노드가
        // 남아있는 경우를 자동으로 감지해 div로 감싸준다 (줄 구조 정리).
        function normalizeEditorStructure() {
            let changed = false;
            Array.from(editor.childNodes).forEach(node => {
                if (node.nodeType === Node.TEXT_NODE) {
                    if (node.textContent.trim() === '') {
                        node.remove();
                    } else {
                        const wrap = document.createElement('div');
                        wrap.textContent = node.textContent;
                        editor.replaceChild(wrap, node);
                    }
                    changed = true;
                }
            });
            return changed;
        }

        let saveTimer, typingEndTimer;
        const processSave = (type) => {
            syncStatus.style.color = "#ff4d4d"; clearTimeout(saveTimer); clearTimeout(typingEndTimer);
            saveTimer = setTimeout(async () => {
                if(isComposing) return;

                        lastSaveDisplay.innerText = formatTime(now);
                    }
                    if(navigator.onLine) syncStatus.style.color = "#4ade80";
                    typingEndTimer = setTimeout(() => { isTyping = false; }, 2000);
                } catch(e) { syncStatus.style.color = "#ff4d4d"; }
            }, 1000);
        };

        function formatTime(timestamp) {
            if(!timestamp) return "저장 기록 없음";
            const n = new Date(timestamp);
            return `${n.getHours()}:${String(n.getMinutes()).padStart(2,'0')}`;
        }

        }

        // [수정된 핵심 로직: 줄간격 교정 복사 기능]
        // 한 줄(div/p) 내부의 텍스트를 추출하되, 줄 내부의 <br>은 \n으로 변환하고
        // innerText 대신 직접 DOM을 순회해서 브라우저별 줄바꿈 중복 문제를 제거한다.
        function getLineText(node) {
            const children = Array.from(node.childNodes);
            let result = '';
            children.forEach((child, idx) => {
                const isLast = (idx === children.length - 1);
                if (child.nodeType === Node.TEXT_NODE) {
                    result += child.textContent;
                } else if (child.nodeName === 'BR') {
                    // 줄(div) 맨 끝에 붙은 <br>은 "빈 줄을 보여주기 위한 자리 표시자"일 뿐,
                    // 실제 개행이 아니므로 무시한다 (이중 개행 방지).
                    // 줄 중간에 있는 <br>만 실제 개행으로 처리한다.
                    if (!isLast) result += '\n';
                } else if (child.nodeName === 'DIV' || child.nodeName === 'P') {
                    // 줄 안에 또 다른 블록 요소가 중첩된 경우 재귀적으로 처리
                    result += (result ? '\n' : '') + getLineText(child);
                } else {
                    result += getLineText(child);
                }
            });
            return result;
        }

        window.copyAllText = () => {
            let textLines = [];
            const childNodes = Array.from(editor.childNodes);

            if (childNodes.length === 0) {
                // 에디터가 비어있거나 div 구조가 전혀 없는 경우
                textLines.push(editor.textContent || "");
            } else {
                childNodes.forEach(node => {
                    if (node.nodeName === 'DIV' || node.nodeName === 'P') {
                        // 각 div(=한 줄)에서 텍스트만 정확히 추출 (빈 줄도 보존)
                        textLines.push(getLineText(node));
                    } else if (node.nodeType === Node.TEXT_NODE) {
                        // 에디터 최상위에 div로 감싸이지 않은 텍스트 노드가 있는 경우
                        textLines.push(node.textContent);
                    } else if (node.nodeName === 'BR') {
                        textLines.push("");
                    }
                });
            }

            // 배열로 담은 라인들을 정확히 1개의 엔터(\n)로만 결합
            const finalCopyText = textLines.join('\n');

            navigator.clipboard.writeText(finalCopyText).then(() => alert("전체 텍스트 복사 완료! (줄간격 교정됨)"));
        };

            // 챕터를 지우거나 순서를 바꾼 이력이 있어도 항상 맨 아래(마지막)에 생성된다.
            const maxOrder = sameFolderChapters.reduce((max, c) => Math.max(max, c.order || 0), -1);
            const order = maxOrder + 1;
            await setDoc(doc(db, "myNovels", id), { title: "새 에피소드", content: "", folderId: currentFolderId, order: order, createdAt: Date.now(), updatedAt: Date.now() });
            loadChapter(id);
        };
